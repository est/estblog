<html>
    <head>
        <title>JS1k, 1k demo submission [856]</title>

    </head>
    <body>
        <canvas id="c"></canvas>
        <script>
            var b = document.body;
            var c = document.getElementsByTagName('canvas')[0];
            var a = c.getContext('2d');
            document.body.clientWidth; // fix bug in chrome.

        <script>
// start of submission //

// end of submission //
        </script>
    </body>
